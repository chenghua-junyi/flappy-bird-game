   <div id="gameContainer">

       <canvas id="gameCanvas"></canvas>

       <div id="scoreDisplay">分數: 0</div>

       <div id="startMessage">

           點擊或按空白鍵開始！

           <button id="startButton">開始遊戲</button>

           <div class="instructions">點擊或按空白鍵來拍動翅膀</div>

       </div>

       <div id="gameOver">

           遊戲結束！

           <div id="finalScore"></div>

           <button id="restartButton">重新開始</button>

       </div>

   </div>

   <script>

       const canvas = document.getElementById('gameCanvas');

       const ctx = canvas.getContext('2d');

       const scoreDisplay = document.getElementById('scoreDisplay');

       const startMessage = document.getElementById('startMessage');

       const startButton = document.getElementById('startButton');

       const gameOverMessage = document.getElementById('gameOver');

       const finalScoreDisplay = document.getElementById('finalScore');

       const restartButton = document.getElementById('restartButton');

       // 遊戲設定

       const GAME_WIDTH = 400;

       const GAME_HEIGHT = 600;

       const BIRD_SIZE = 30;

       const GRAVITY = 0.5;

       const JUMP_STRENGTH = -8;

       const PIPE_WIDTH = 60;

       const PIPE_GAP = 150; // 管道之間的垂直間隙

       const PIPE_SPEED = 3;

       const GROUND_HEIGHT = 50;

       let bird = {

           x: 50,

           y: GAME_HEIGHT / 2 - BIRD_SIZE / 2,

           velocity: 0

       };

       let pipes = [];

       let score = 0;

       let gameStarted = false;

       let gameOver = false;

       let animationFrameId;

       // 調整畫布大小以適應容器

       function resizeCanvas() {

           const container = document.getElementById('gameContainer');

           canvas.width = container.clientWidth;

           canvas.height = container.clientHeight;

           // 重新計算鳥的位置，以防萬一

           if (!gameStarted) {

               bird.y = canvas.height / 2 - BIRD_SIZE / 2;

           }

       }

       window.addEventListener('resize', resizeCanvas);

       resizeCanvas(); // 初始調整大小

       // 繪製鳥

       function drawBird() {

           ctx.fillStyle = 'yellow';

           ctx.beginPath();

           ctx.arc(bird.x + BIRD_SIZE / 2, bird.y + BIRD_SIZE / 2, BIRD_SIZE / 2, 0, Math.PI * 2);

           ctx.fill();

           ctx.strokeStyle = 'orange';

           ctx.lineWidth = 2;

           ctx.stroke();

           // 簡單的眼睛

           ctx.fillStyle = 'black';

           ctx.beginPath();

           ctx.arc(bird.x + BIRD_SIZE * 0.7, bird.y + BIRD_SIZE * 0.3, BIRD_SIZE * 0.1, 0, Math.PI * 2);

           ctx.fill();

           // 簡單的嘴巴

           ctx.fillStyle = 'orange';

           ctx.beginPath();

           ctx.moveTo(bird.x + BIRD_SIZE, bird.y + BIRD_SIZE / 2);

           ctx.lineTo(bird.x + BIRD_SIZE + 10, bird.y + BIRD_SIZE / 2 - 5);

           ctx.lineTo(bird.x + BIRD_SIZE + 10, bird.y + BIRD_SIZE / 2 + 5);

           ctx.closePath();

           ctx.fill();

       }

       // 繪製管道

       function drawPipes() {

           ctx.fillStyle = 'green';

           ctx.strokeStyle = 'darkgreen';

           ctx.lineWidth = 3;

           pipes.forEach(pipe => {

               // 頂部管道

               ctx.fillRect(pipe.x, 0, PIPE_WIDTH, pipe.topHeight);

               ctx.strokeRect(pipe.x, 0, PIPE_WIDTH, pipe.topHeight);

               // 底部管道

               ctx.fillRect(pipe.x, pipe.bottomY, PIPE_WIDTH, canvas.height - pipe.bottomY - GROUND_HEIGHT);

               ctx.strokeRect(pipe.x, pipe.bottomY, PIPE_WIDTH, canvas.height - pipe.bottomY - GROUND_HEIGHT);

           });

       }

       // 繪製地面

       function drawGround() {

           ctx.fillStyle = '#D2B48C'; /* 棕色 */

           ctx.fillRect(0, canvas.height - GROUND_HEIGHT, canvas.width, GROUND_HEIGHT);

           ctx.strokeStyle = '#8B4513'; /* 深棕色 */

           ctx.lineWidth = 3;

           ctx.strokeRect(0, canvas.height - GROUND_HEIGHT, canvas.width, GROUND_HEIGHT);

       }

       // 更新遊戲狀態

       function update() {

           if (gameOver || !gameStarted) return;

           // 更新鳥的位置

           bird.velocity += GRAVITY;

           bird.y += bird.velocity;

           // 檢查鳥是否超出邊界

           if (bird.y + BIRD_SIZE > canvas.height - GROUND_HEIGHT || bird.y < 0) {

               endGame();

               return;

           }

           // 移動管道並生成新管道

           if (pipes.length === 0 || pipes[pipes.length - 1].x < canvas.width - 200) {

               generatePipe();

           }

           pipes.forEach(pipe => {

               pipe.x -= PIPE_SPEED;

               // 檢查碰撞

               if (

                   bird.x < pipe.x + PIPE_WIDTH &&

                   bird.x + BIRD_SIZE > pipe.x &&

                   (bird.y < pipe.topHeight || bird.y + BIRD_SIZE > pipe.bottomY)

               ) {

                   endGame();

                   return;

               }

               // 增加分數

               if (!pipe.passed && bird.x > pipe.x + PIPE_WIDTH) {

                   score++;

                   scoreDisplay.textContent = `分數: ${score}`;

                   pipe.passed = true;

               }

           });

           // 移除超出螢幕的管道

           pipes = pipes.filter(pipe => pipe.x + PIPE_WIDTH > 0);

       }

       // 生成新管道

       function generatePipe() {

           const minHeight = 50;

           const maxHeight = canvas.height - GROUND_HEIGHT - PIPE_GAP - minHeight;

           const topHeight = Math.floor(Math.random() * (maxHeight - minHeight + 1)) + minHeight;

           const bottomY = topHeight + PIPE_GAP;

           pipes.push({

               x: canvas.width,

               topHeight: topHeight,

               bottomY: bottomY,

               passed: false

           });

       }

       // 遊戲循環

       function gameLoop() {

           ctx.clearRect(0, 0, canvas.width, canvas.height); // 清除畫布

           update();

           drawPipes();

           drawGround();

           drawBird();

           if (!gameOver) {

               animationFrameId = requestAnimationFrame(gameLoop);

           }

       }

       // 拍動翅膀

       function flap() {

           if (!gameOver) {

               bird.velocity = JUMP_STRENGTH;

           }

       }

       // 開始遊戲

       function startGame() {

           bird = {

               x: 50,

               y: canvas.height / 2 - BIRD_SIZE / 2,

               velocity: 0

           };

           pipes = [];

           score = 0;

           scoreDisplay.textContent = `分數: ${score}`;

           gameOver = false;

           gameStarted = true;

           startMessage.style.display = 'none';

           gameOverMessage.style.display = 'none';

           if (animationFrameId) {

               cancelAnimationFrame(animationFrameId);

           }

           gameLoop();

       }

       // 結束遊戲

       function endGame() {

           gameOver = true;

           gameStarted = false;

           cancelAnimationFrame(animationFrameId);

           finalScoreDisplay.textContent = `你的分數是: ${score}`;

           gameOverMessage.style.display = 'flex';

       }

       // 事件監聽器

       document.addEventListener('keydown', (e) => {

           if (e.code === 'Space') {

               if (!gameStarted && !gameOver) {

                   startGame();

               } else {

                   flap();

               }

           }

       });

       canvas.addEventListener('click', () => {

           if (!gameStarted && !gameOver) {

               startGame();

           } else {

               flap();

           }

       });

       startButton.addEventListener('click', startGame);

       restartButton.addEventListener('click', startGame);

       // 初始顯示開始訊息

       startMessage.style.display = 'flex';

   </script>
